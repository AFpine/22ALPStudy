#include<iostream>
using namespace std;
int main() {
	int N;
	cin >> N;
	int num[1000007] = { 0 };
	int arr[100005] = { 0 };
	int sum = 0;
	for (int i = 1; i <= N; i++) {
		cin >> arr[i];
		num[arr[i]] = i;
	}
	int X;
	cin >> X;
	for (int i = 1; i <= N; i++) {
		if ((X - arr[i]) < 1000001) {
			if (X >= arr[i] && i < num[X - arr[i]]) {
				sum++;
			}
		}
	}
	cout << sum;
}
